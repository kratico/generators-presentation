Generators as Observers

interface Observer {
  next(value? : any) : void;
  return(value? : any) : void;
  throw(error) : void;
}

Generators as Observers

As an observer, a generator pauses until it receives input.

There are three kinds of input, transmitted via the methods specified by the interface:

  • next() sends normal input.
  • return() terminates the generator.
  • throw() signals an error.
function* dataConsumer() {
    console.log('Started');
    console.log('1. received', yield);
    console.log('2. received', yield);
    return 'result';
}       
> var genObj = dataConsumer();     
> genObj.next() // starts the observer
Started
{ value: undefined, done: false }
> genObj.next('a')
1. received a
{ value: undefined, done: false }
> genObj.next('b')
2. received b
{ value: 'result', done: true }

return() and throw()

  • return(x) executes return x at the location of yield.
  • throw(x) executes throw x at the location of yield.

return()

function* genFunc() {
  try {
    console.log('Started');
    yield; // (A)
  } finally {
    console.log('Exiting');
  }
}
> var genObj = genFunc();
> genObj.next()
Started
{ value: undefined, done: false }
> genObj.return('Result')
Exiting
{ value: 'Result', done: true }

throw()

function* genFunc() {
  try {
    console.log('Started');
    yield; // (A)
  } catch (error) {
    console.log('Caught: ' + error);
  }
}
> var genObj = genFunc();
> genObj.next()
Started
{ value: undefined, done: false }
> genObj.throw(new Error('Problem!'))
Caught: Error: Problem!
{ value: undefined, done: true }

Why do we need a runner?

function* genFunc() {
  yield Promise.resolve(1);
}
> var genObj = genFunc();
> genObj.next().value.then(function(value) {
    console.log('value:', value);
});
value: 1

function co(genFunc) {
  var genObj = genFunc();
  run();
  function run(promiseResult) {
    var result = genObj.next(promiseResult);
    if (!result.done) {
      // A Promise was yielded
      result.value
        .then(function(result) {
          run(result);
        })
        .catch(function(error) {
          genObj.throw(error);
        });
    }
  }
}

Error Handling

Callback Request + Error

function request(url, cb) {              
  $.get(url, function(result) {
    cb(null, result);
  })
  .fail(function(jqXHR, status, error) {
    cb(error);
  });
}

Promise Request + Error

function request(url) {
  return new Promise(function(resolve, reject) {
    $.get(url, resolve)
    .fail(function(jqXHR, status, error) {
      reject(error);
    });
  });
}

Generator Error handling

co(function* getUsersForTweets() {
  var users = [];
  try {
    var tweets = yield getTweets();         // A
    var userIds = getUniqueUserIds(tweets); // B
    users = yield userIds.map(getUser);     // C
  } catch (ex) {
    console.error(ex);
  }
  return users;
})(function(err, users) {
  if (err) return console.error(err);
  console.log(users);
});
